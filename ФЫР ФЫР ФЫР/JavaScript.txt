		массивы
var cats = ["timka", "kesha", "sobaka"];  - массив
var dogs = ["graf", "fyntic", "cot"];  - массив
console.log(cats);
console.log(cats[1]);  - вызов первого элемента массива
cats[1] = "qwerty";  - есть ли это слово
console.log(cats.length);  - длина 
console.log(cats[cats.length - 1]);  - последний элемент
console.log(cats.push("may"));  - добавить элемент в конец
console.log(cats.unshift("tochnoCot"));  - добавить элемент в начало
console.log(cats.pop());  - удалить последний элемент
console.log(cats.shift());  - удалить первый элемент
var all = cats.concat(dogs);  - соединяет 2 массива
console.log(cats.indexOf("sobaka"));  - возвращает первый индекс найденного элемента
console.log(cats.lastIndexOf("sobaka"));  - возвращает последний индекс найденного элемента
console.log(cats.join(" - "));  - ставит между всеми элементами массива " - "
